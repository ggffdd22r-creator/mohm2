    <script>
        // اختيار سرعة الإنترنت
        const speedOptions = document.querySelectorAll('.speed-option');
        const selectedSpeedInput = document.getElementById('selectedSpeed');
        
        speedOptions.forEach(option => {
            option.addEventListener('click', function() {
                // إزالة التحديد من جميع الخيارات
                speedOptions.forEach(opt => opt.classList.remove('selected'));
                
                // إضافة التحديد للخيار المختار
                this.classList.add('selected');
                
                // حفظ القيمة المختارة
                const speed = this.getAttribute('data-speed');
                selectedSpeedInput.value = speed;
                
                // إضافة تأثير بسيط
                this.style.transform = 'scale(1.05)';
                setTimeout(() => {
                    this.style.transform = 'scale(1)';
                }, 300);
            });
        });
        
        // اختيار افتراضي
        speedOptions[1].click();
        
        // معالجة نموذج الدخول
        document.getElementById('loginForm').addEventListener('submit', function(e) {
            e.preventDefault();
            
            const username = document.getElementById('username').value;
            const speed = selectedSpeedInput.value;
            
            if (!username || !speed) {
                alert('يرجى إدخال اسم المستخدم واختيار السرعة');
                return;
            }
            
            // تأثير تحميل
            const submitBtn = document.querySelector('.btn-primary');
            const originalText = submitBtn.innerHTML;
            submitBtn.innerHTML = '<i class="fas fa-spinner fa-spin"></i> جارٍ الدخول...';
            submitBtn.disabled = true;
            
            // محاكاة اتصال بالخادم
            setTimeout(() => {
                // هنا يمكنك إضافة كود الاتصال بالخادم (API)
                alert(`مرحباً ${username}! تم تسجيل دخولك بنجاح.\nسرعة الإنترنت المختارة: ${speed} Mbps`);
                
                // إعادة زر الدخول لحالته الأصلية
                submitBtn.innerHTML = originalText;
                submitBtn.disabled = false;
                
                // في الواقع، هنا سيتم توجيه المستخدم إلى الصفحة الرئيسية
                // window.location.href = 'dashboard.html';
            }, 1500);
        });
        
        // تأثير عند التركيز على حقل الإدخال
        const inputs = document.querySelectorAll('.form-control');
        inputs.forEach(input => {
            input.addEventListener('focus', function() {
                this.parentElement.querySelector('i').style.color = '#3498db';
            });
            
            input.addEventListener('blur', function() {
                this.parentElement.querySelector('i').style.color = '#bdc3c7';
            });
        });
        
        // إضافة تأثيرات إضافية
        window.addEventListener('load', function() {
            const logoIcon = document.querySelector('.logo i');
            logoIcon.style.transition = 'transform 0.5s';
            
            setInterval(() => {
                logoIcon.style.transform = 'rotate(10deg)';
                setTimeout(() => {
                    logoIcon.style.transform = 'rotate(-10deg)';
                }, 500);
                setTimeout(() => {
                    logoIcon.style.transform = 'rotate(0deg)';
                }, 1000);
            }, 5000);
        });
    </script>
